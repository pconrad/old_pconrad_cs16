   (*cPtr).b=bVal;
}

  3. struct Complex squareComplex(struct Complex c){
   struct Complex ans;
   initComplex(&ans, (c.a*c.a - c.b*c.b), ( 2*c.a*c.b));
   return ans;
}

         printNChars(w/2,' ');
